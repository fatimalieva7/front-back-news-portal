        document.addEventListener('DOMContentLoaded', function() {
            if(isLoggedIn) {
                document.getElementById('guest-buttons').style.display = 'none';
                document.getElementById('user-buttons').style.display = 'block';
            }
        });
        
        // Функция для переключения состояния (только для демонстрации)
        function toggleAuth() {
            isLoggedIn = !isLoggedIn;
            if(isLoggedIn) {
                document.getElementById('guest-buttons').style.display = 'none';
                document.getElementById('user-buttons').style.display = 'block';
            } else {
                document.getElementById('guest-buttons').style.display = 'block';
                document.getElementById('user-buttons').style.display = 'none';
            }
        }
